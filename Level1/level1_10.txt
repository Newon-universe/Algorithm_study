import java.util.Scanner

fun main() = with(Scanner(System.`in`)) {
    
    val A = nextInt()
    val B = nextInt()
    val C = nextInt()
    
    var result1 = (A+B)%C
    var result2 = ((A%C) + (B%C))%C
    var result3 = (A*B)%C
    var result4 = ((A%C) * (B%C))%C
// 이렇게 따로 변수에 넣어줄 필요는 없지만 문제가 '(A+B)%C 는 ((A%C) + (B%C))%C 와 같을까?' 라는 말이 붙어있길래 넣어봄. 만약 문제 조건이 비교연산 이었다면 이 방법이 코드 가독성은 더 좋기 때문.
    
    println(result1)
    println(result2)
    println(result3)
    println(result4)
}
