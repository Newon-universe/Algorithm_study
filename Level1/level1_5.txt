import java.util.Scanner

fun main(args: Array<String>) {
    val sc: Scanner = Scanner(System.`in`) // kotlin에서 in은 키워드이기 때문에 저렇게 써줘야 한다고 한다. 자바랑 달랐다.
    var A = sc.nextInt() // Scanner 객체 앞에 with 문을 쓰면 sc 프로퍼티를 사용하지 않고도 nextInt() 메서드를 불러올 수 있다고 한다. with문에 대해서는 공부해 봐야 될 것 같다.
    var B = sc.nextInt()
    
    if(0 < A && B < 10) { // if문 조건체크 때문에 시간은 더 오래 걸렸다. 사실 알고리즘 문제는 입력예제가 정해져 있어서 이렇게 단순한 연산의 경우 굳이 if문을 붙여 시간복잡도를 높일 필요는 없었지만 if문을 안 써줬을 때와의 차이를 비교해보기 위해 그냥 써봤다. 약 20ms 정도 더 걸렸다.
        println(A+B)
    }
}
