import java.util.Scanner

fun main() = with(Scanner(System.`in`)) {
    
    var x = nextInt()
    var y = nextInt()
    
    if ( x > 0 && y > 0) {
        println("1")
    }
    else if ( x < 0 && y > 0) {
        println("2")
    }
    else if ( x < 0 && y < 0) {
        println("3")
    }
    else {
        println("4")
    }
}
