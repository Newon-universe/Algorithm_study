// 컴파일 에러난 코드

import java.util.Scanner

fun main():Double = with(Scanner(System.`in`)) {
    println(nextInt()/nextInt()) // nextInt() 로 Int 타입을 입력받아 놓고 반환타입을 Double로 받았다. 당연히 type mismatch 에러가 났다. Int 연산을 핸호고 어떻게 Double 타입으로 값을 받냐는 에러였다. 입력예제가 정수고 출력예제가 실수여서 생각을 이상하게 했다.
}

// 정답 코드

import java.util.Scanner

fun main() = with(Scanner(System.`in`)) {
    println(nextDouble()/nextDouble()) // 애초에 연산을 Double 타입으로 해줘야 반환값도 Double 타입 실수로 잘 받는다.
}
